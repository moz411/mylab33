    def PCA(X):
    res = {'eig_vecs':0,'eig_vals':0,'ind':0}
    #  matrice de covariance
    cov_mat = np.cov(X.T)
    # eigen decomposition
    res['eig_vals'], res['eig_vecs'] = np.linalg.eig(cov_mat)
    # tri ordre dÃ©croissant
    idx = np.argsort(res['eig_vals'])[::-1]
    res['eig_vecs'] = res['eig_vecs'][:,idx]
    res['eig_vals'] = res['eig_vals'][idx]
    # Distance des individus
    res['ind'] = np.dot(res['eig_vecs'].T, X.T).T
    # Conversion en Dataframes
    for key in res.keys(): res[key] = pd.DataFrame(res[key]) 
    res['eig_vals'] = res['eig_vals'].T # petit bug Pandas :-)
    for key in res.keys(): res[key].columns = list('PC%d' % i for i in range(1,res[key].shape[1]+1))
    res['eig_vecs'].index = X.columns
    # Ratio de la variance expliquÃ©e
    res['eig_vals_ratio'] = round((res['eig_vals'] / sum(res['eig_vals'].values[0])) * 100,2)
    # QualitÃ© de la reprÃ©sentation des individus
    # Calcul du cosinus carrÃ©
    res['cos2'] = pd.DataFrame()
    for i in range(1,res['ind'].shape[1]):
        PC1 = 'PC%d' % i; PC2 = 'PC%d' % (i+1)
        res['cos2'][PC1] = res['ind'].apply(lambda row:np.cos(row[PC1]/np.sqrt((row[PC1]+row[PC2])**2))**2,axis=1)
    # Contribution des individus en %
    res['contrib'] = pd.DataFrame()
    tot = res['ind'].shape[0]
    for i in range(1,res['ind'].shape[1]+1):
        PC = 'PC%d' % i
        res['contrib'][PC] = res['ind'].apply(lambda row:row[PC]**2/(res['eig_vals'][PC][0]*tot)*1e3,axis=1)

    return res

    def cleanup(data):
    data = pd.DataFrame(data.split())
    data = data[0].str.split(',',expand=True)
    data.columns = data.iloc[0]
    data = data.drop(0).reset_index(drop=True)
